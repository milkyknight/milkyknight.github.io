/*
Lucas定理求组合数 C(n,m)
1.mod为素数,mod不为素数考虑exgcd
2.调用方式： lucas(n,m);
/
const int mod = 1e9 + 7;
ll quick_pow(ll a,ll m){
ll ans = 1;
while(m){
if(m&1) ans = (ans * a)%mod;
a = (aa)%mod;
m >>= 1;
}
return ans;
}
ll inv(ll x){return quick_pow(x, mod-2);}
ll comb(ll n,ll m){
if(m>n) return 0;
ll up = 1;ll down = 1;
for(int i = n-m+1;i<=n;i++) up = (upi)%mod;
for(int i=1;i<=m;i++) down = (downi)%mod;
return up*inv(down)%mod;
}
ll lucas(ll n,ll m){
if(m==0) return 1;
return comb(n%mod,m%mod)*lucas(n/mod,m/mod)%mod;
}

/*
krusal求最小生成树
*/
const int maxn = 1e5;
int f[maxn];
int size[maxn];
struct EGE{
int from,to,w;
EGE (int a,int b,int c): from(a),to(b),w(c){}
EGE(){}
};EGE edge[maxn];
int find(int x){
if(x == f[x]) return x;
f[x] = find(f[x]);
return f[x];
}
void join(int fx,int fy){
if(size[fx] > size[fy]) swap(fx,fy);
f[fx] = fy; size[fy] += size[fx];
}
bool cmp(EGE a,EGE b){return a.w < b.w;}
void krusal()
{
int n,m,x,y,z;
cin >> n >> m;
for(int i=1;i<=n;i++){
f[i] = i;
size[i] = 1;
}
for(int i=1;i<=m;i++){
cin >> x >> y >> z;
edge[i] = {x,y,z};
}
sort(edge+1,edge+m+1,cmp);
int road = n-1; ll cost = 0;
for(int i=1;i<=m;i++){
int fx = find(edge[i].from);
int fy = find(edge[i].to);
if(fx == fy) continue;
join(fx,fy);
road–; cost += edge[i].w;
if(road==0) break;
}
if(road) cout << “orz”;
else cout << cost;
}

/*
Dijkstra

vector g[maxn];
int dis[maxn];
void dijkstra(int s){
priority_queue<pii,vector,greater > pq;
memset(dis, 0x3f3f3f3f, sizeof(dis));
dis[s] = 0;
pq.push(pii(0,s));
while(!pq.empty()){
pii now = pq.top(); pq.pop();
int u = now.second;
if(now.first > dis[u]) continue;
for(int i=0;i<g[u].size();i++){
int v = g[u][i].to;
if(dis[v] > g[u][i].w + dis[u]){
dis[v] = g[u][i].w + dis[u];
pq.push(pii(dis[v],v));
}
}
}
}
/
shortpath in DAG 可以解决有向无环图上带负权的最短路，dp的思想
注意:如果多组数据 vector 记得 clear() cnt 归零

int topo[maxn],vis[maxn],cnt = 0;
//vector g[maxn];
void dfs(int s){ // dfs求拓扑序列
vis[s] = 1;
for(int i=0;i<g[s].size();i++){
int to = g[s][i].to;
if(!vis[to]) dfs(to);
}
topo[cnt++] = s;
}
void dagSP(int s){
int i = cnt;
int dis[maxn];
memset(dis, 0x3f, sizeof(dis));
while(topo[–i]!=s); //在拓扑序列中找到s的位置
for(;i>=0;i–){
int u = topo[i];
for(int j=0;j<g[u].size();j++){
int v = g[u][j].to;
dis[v] = min(dis[v],dis[u] + g[u][j].w);
}
}
}
